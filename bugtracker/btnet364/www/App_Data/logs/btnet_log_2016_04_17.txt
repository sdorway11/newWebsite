isnull(ba.bp_file,'') ba_file,  -- intentionally ba

left outer join bug_posts ba on ba.bp_parent = a.bp_id and ba.bp_bug = a.bp_bug

 and a.bp_type <> 'update'
 and a.bp_hidden_from_external_users = 0
